172 static void *_alloc( Arena **p, size_t mem_sz )
173 {
174  // First reject anything nonsensical or excessively large .
175  if ( mem_sz == 0 || mem_sz > PTRDIFF_MAX ) {
176  return NULL; // request impossibly large (out of memory)
177  }
178  // Convert to signed so that it does not implicitly cast expression
179  // using it to size_t later. After this mem_sz is no longer needed.
180  ptrdiff_t mem_pd = mem_sz;
181  // Padding will be a small non-negative number. It is always safe
182  // to subtract from a size, as the result will at worst be a small
183  // negative number.
184  ptrdiff_t padding = -mem_pd & ( MAX_ALIGN - 1 );
185 
186 
187  Arena *ap;
188  for ( ap = *p;; *p = ap ) {
189  // Work using a size, not with pointer arithmetic.
190  ptrdiff_t available = ap->end - ap->begin;
191 
192  // Per the note, this subtraction is safe. Both operands are
193  // signed, so no surprise promotions turning a small negative
194  // into large positive (i.e. size_t).
195  if ( mem_pd > available - padding ) {
196  if ( ap->next ) {
197  ap = ap->next;
198  ap->begin = ( char * ) ap + AHS; // reset?
199  continue; // keep looking
200 
201  } else { // End of the list, allocate a new chunk.
202  // It is *not* yet safe to add header_size to mem_pd,
203  // so subtract from the other side.
204  if ( mem_pd > PTRDIFF_MAX - AHS - padding ) {
205  return NULL; // request too large for metadata
206  }
207  // At this point we know header_size+mem_pd+padding is
208  // safe to compute.
209 
210  // Note: chunk_sz does not require any alignment padding.
211  // That's already built into header_size.
212  ptrdiff_t real_size = MAX( (mem_pd + padding), (ptrdiff_t) (*p)->chunk_sz );
213 
214  ptrdiff_t chunk_sz = ap->chunk_sz; // save a copy
215  ap = ap->next = malloc( AHS + real_size );
216  if ( !ap ) {
217  return NULL; // OOM (can happen on Linux with huge mem_sz!)
218  }
219  ap->next = NULL;
220  ap->chunk_sz = chunk_sz;
221  ap->begin = ( char * ) ap + AHS;
222  ap->end = ap->begin + real_size;
223  break; // use this arena
224  }
225  } else {
226  break; // found space
227  }
228  }
229 
230  void *ptr = ap->begin;
231  ap->begin += mem_pd + padding; // checks passed, so addition is safe
232  // Note: I strongly recommend zero-initialization by default. It
233  // makes for clearer, shorter, and more robust programs. If it's
234  // too slow in some cases, add an opt-out flag.
235 
236  // Zeroing out the last byte and the padding
237  // will implement an `arena_calloc` that zeroes out all returned memory,
238  // to follow the usual convention, to get it done! makes swapping methods
239  // easier.
240  for ( char *zptr = ap->begin - ( padding + 1 ); zptr < ap->begin; zptr++ ) {
241  *zptr = '\0';
242  }
243  return ptr;
244  // return memset(ptr, 0, mem_pd);
245 }

255 void *arena_alloc( int n, size_t mem_sz )
256 {
257  const char *emsg = "arena_alloc: Couldn't allocate memory for arena with mem_pd: %lu.\n" "Aborting.";
258  // First reject anything nonsensical or excessively large .
259  if ( mem_sz == 0 || mem_sz > PTRDIFF_MAX ) {
260  return NULL; // request impossibly large (out of memory)
261  }
262 
263  void *p;
264  p = arenas[n]->begin;
265  ptrdiff_t mem_pd = mem_sz;
266  ptrdiff_t padding = -mem_pd & ( MAX_ALIGN - 1 );
267  mem_pd += padding ;
268  if ( mem_pd < ( ptrdiff_t ) mem_sz || ( ( mem_pd + AHS ) > PTRDIFF_MAX ) ) {
269  fprintf( stderr,emsg , (size_t) mem_pd );
270  abort( ); // Overflow conditions.
271  }
272  if ( ( arenas[n]->begin += mem_pd ) > arenas[n]->end ) {
273  // padding is already added to mem_pd here.
274  p = _alloc( &arenas[n], mem_sz );
275  } else { // zero out last byte and padding
276  for ( char *zptr = arenas[n]->begin - ( padding + 1 ); zptr < arenas[n]->begin; zptr++ ) {
277  *zptr = '\0';
278  }
279  }
280 
281  return p;
282 }
