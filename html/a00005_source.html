226 static Arena *_arena_init( size_t chunk_sz )
227 {
228  static const char *emsg = "_arena_init: The chunk_sz requested is to small: %d\n";
229  static const char *emsg2 = "_arena_init: The chunk_sz: %lu requested is too large.\n"
230  "The request is larger than ARENAS_MAX_ALLOC %lu: %d ";
231 
232  if ( chunk_sz < MALLOC_PTR_SIZE ) {
233  fprintf( stderr, emsg, chunk_sz );
234  abort( );
235  }
236  chunk_sz -= MALLOC_PTR_SIZE;
237  if ( chunk_sz < MAX_ALIGN ) {
238  fprintf( stderr, emsg, chunk_sz );
239  abort( );
240  }
241  // the size of the pointer malloc uses to address the allocated block.
242  size_t padding = -chunk_sz & ( MAX_ALIGN - 1 );
243 
244  if ( padding ) {
245  chunk_sz -= ( MAX_ALIGN - padding ); // Guarranteed to be a positive
246  // number.
247  }
248 
249  if ( chunk_sz <= ( size_t ) _AHS ) {
250  fprintf( stderr, emsg, ( chunk_sz + padding + sizeof( unsigned int ) ) );
251  abort( );
252  }
253 
254  if ( chunk_sz > ARENAS_MAX_ALLOC ) {
255  fprintf( stderr, emsg2, chunk_sz, ARENAS_MAX_ALLOC );
256  abort( );
257  }
258 
259  Arena *p;
260  p = malloc( chunk_sz );
261  if ( !p ) {
262  return NULL;
263  }
264  p->chunk_sz = chunk_sz;
265  p->begin = ( char * ) p + _AHS;
266  p->end = ( char * ) p + chunk_sz; // real_size;
267  p->next = NULL;
268  // see https://nullprogram.com/blog/2023/09/27/ (the alloca() function //
269  return p;

289 static void *_alloc( Arena ** p, size_t mem_sz, size_t n )
290 {
291  // First reject anything nonsensical or excessively large .
292  if ( mem_sz == 0 || mem_sz > PTRDIFF_MAX ) {
293  return NULL; // request impossibly large (out of memory)
294  }
295  // Convert to signed so that it does not implicitly cast expression
296  // using it to size_t later. After this mem_sz is no longer needed.
297  ptrdiff_t mem_pd = mem_sz;
298  // Padding will be a small non-negative number. It is always safe
299  // to subtract from a size, as the result will at worst be a small
300  // negative number.
301  ptrdiff_t padding = -mem_pd & ( MAX_ALIGN - 1 );
302 
303 
304  Arena *ap;
305  for ( ap = *p;; *p = ap ) {
306  // Work using a size, not with pointer arithmetic.
307  ptrdiff_t available = ap->end - ap->begin;
308  assert(available > 0 );
309 
310  // Per the note, this subtraction is safe. Both operands are
311  // signed, so no surprise promotions turning a small negative
312  // into large positive (i.e. size_t).
313  if ( mem_pd > available - padding ) {
314  if ( ap->next ) {
315  ap = ap->next;
316  ap->begin = ( char * ) ap + _AHS; // reset?
317  continue; // keep looking
318 
319  } else { // End of the list, allocate a new chunk.
320  // It is *not* yet safe to add header_size to mem_pd,
321  // so subtract from the other side.
322  if ( mem_pd > (PTRDIFF_MAX - _AHS - padding) ) {
323  return NULL; // request too large for metadata
324  }
325  // At this point we know header_size+mem_pd+padding is
326  // safe to compute.
327 
328  // Note: chunk_sz does not require any alignment padding,
329  // accounted for.
330  ptrdiff_t real_size = MAX( ( mem_pd + padding + _AHS ), ( ptrdiff_t ) ( *p )->chunk_sz );
331 
332  ptrdiff_t chunk_sz = ap->chunk_sz; // save a copy
333  fprintf(stderr,"REAL_SIZE: %ld\n",real_size) ;
334  ap = ap->next = malloc( real_size );
335  if ( !ap ) {
336  return NULL; // OOM (can happen on Linux with huge mem_sz!)
337  }
338 #if ARENAS_LOG_LEVEL > 0
339  allocated_chunks[n] += real_size;
340  allocation_chunk_count[n] += 1 ;
341 #endif
342  ap->next = NULL;
343  ap->chunk_sz = chunk_sz;
344  ap->begin = ( char * ) ap + _AHS;
345  ap->end = ap->begin + real_size;
346  break; // use this arena
347  }
348  } else {
349  break; // found space
350  }
351  }
352 
353  void *ptr = ap->begin;
354  ap->begin += mem_pd + padding; // checks passed, so addition is safe
355  // Note: I strongly recommend zero-initialization by default. It
356  // makes for clearer, shorter, and more robust programs. If it's
357  // too slow in some cases, add an opt-out flag.
358 
359  // Zeroing out the last byte and the padding
360  // will implement an `arena_calloc` that zeroes out all returned memory,
361  // to follow the usual convention, to get it done! makes swapping methods
362  // easier.
363  for ( char *zptr = ap->begin - ( padding + 1 ); zptr < ap->begin; zptr++ ) {
364  *zptr = '\0';
365  }
366  return ptr;
367  // return memset(ptr, 0, mem_pd);
368 }

420 void *arena_alloc( size_t n, size_t mem_sz )
421 {
422  if ( n >= ARENAS_MAX ) {
423  fprintf( stderr, msgBadArena, n, ARENAS_MAX );
424  abort( ); // Overflow conditions.
425  }
426 
427  static const char *emsg = "arena_alloc: Couldn't allocate memory for arena with mem_pd: %lu.\n" ;
428  // First reject anything nonsensical or excessively large .
429  if ( mem_sz == 0 || mem_sz > PTRDIFF_MAX ) {
430  return NULL; // request impossibly large (out of memory)
431  }
432 
433  void *p;
434  p = arenas[n]->begin;
435  ptrdiff_t mem_pd = mem_sz;
436  ptrdiff_t padding = -mem_pd & ( MAX_ALIGN - 1 );
437  mem_pd += padding;
438  if ( mem_pd < ( ptrdiff_t ) mem_sz || ( mem_pd > (PTRDIFF_MAX - _AHS) ) ) {
439  fprintf( stderr, emsg, ( size_t ) mem_pd );
440  abort( ); // Overflow conditions.
441  }
442  ptrdiff_t gauge = (ptrdiff_t)arenas[n]->begin + mem_pd ;
443  assert( gauge >= 0) ;
444  if ( gauge >= (ptrdiff_t) arenas[n]->end ) {
445  // padding is already added to mem_pd here.
446  p = _alloc( &arenas[n], mem_sz, n );
447  } else { // zero out last byte and padding
448  arenas[n]->begin += mem_pd ;
449  for ( char *zptr = arenas[n]->begin - ( padding + 1 ); zptr < arenas[n]->begin; zptr++ ) {
450  *zptr = '\0';
451  }
452  }
453 
454 #if ARENAS_LOG_LEVEL > 1
455  allocated_memory[n] += mem_sz;
456  allocation_memory_count[n] += 1;
457 #endif
458 
459  return p;
460 }
