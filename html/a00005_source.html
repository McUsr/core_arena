382 static Arena *_arena_init( size_t n, size_t chunk_sz )
383 {
384  ptrdiff_t chunk_pd = chunk_sz; // maybe someone without gcc wants to compile it.
385 
386  if ( chunk_pd < MALLOC_PTR_SIZE + MAX_ALIGN ) {
387  fprintf( stderr, alloc_emsg, chunk_sz );
388  abort( );
389  }
390  chunk_pd -= MALLOC_PTR_SIZE;
392  // the size of 1024, but I think malloc will add 8 to 1024.
393  // so for instance the smart size to ask for is 4096-8 == 4088.
394 
395  // the size of the pointer malloc uses to address the allocated block.
396  ptrdiff_t padding = -chunk_pd & ( MAX_ALIGN - 1 );
397 
398  if ( padding ) {
399  chunk_pd -= ( MAX_ALIGN - padding ); // Guaranteed to be positive.
400  }
401 
402  if ( chunk_pd <= ( ptrdiff_t ) _AHS ) {
403  fprintf( stderr, alloc_emsg,"_arena_init", ( chunk_pd + padding + MALLOC_PTR_SIZE ) );
405  abort( );
406  }
407 
408  if ( chunk_pd > (ssize_t) (ARENAS_MAX_ALLOC - MALLOC_PTR_SIZE) ) {
409  fprintf( stderr, alloc_emsg2, "_arena_init", chunk_pd, ARENAS_MAX_ALLOC );
410  abort( );
411  } else if ( tot_mem_usage > ARENAS_MAX_ALLOC - (chunk_pd + MALLOC_PTR_SIZE) ) {
412  fprintf( stderr, alloc_emsg3, "_arena_init",chunk_pd, ARENAS_MAX_ALLOC );
413  abort( );
414  }
415 
416  Arena *p;
417  p = malloc( chunk_pd );
418  if ( !p ) {
419  return NULL;
420  }
421  tot_mem_usage += chunk_pd ; // updates total allocated.
422  if ( chunk_pd < _128K ) {
423  arenas_mem_malloced[n] += chunk_pd ;
424  } else {
425  arenas_mem_mmapped[n] += chunk_pd ;
426  }
427  p->chunk_sz = chunk_pd;
428  p->begin = ( char * ) p + _AHS;
429  p->end = ( char * ) p + chunk_pd; // real_size;
430  p->next = NULL;
431  // see https://nullprogram.com/blog/2023/09/27/ (the alloca() function //
432  return p;
433 }

453 static void *_alloc( Arena ** p, size_t mem_sz, size_t n )
454 {
455  // First reject anything nonsensical or excessively large .
456  if ( mem_sz == 0 || mem_sz > PTRDIFF_MAX ) {
457  return NULL; // request impossibly large (out of memory)
458  }
459  // Convert to signed so that it does not implicitly cast expression
460  // using it to size_t later. After this mem_sz is no longer needed.
461  ptrdiff_t mem_pd = mem_sz;
462  // Padding will be a small non-negative number. It is always safe
463  // to subtract from a size, as the result will at worst be a small
464  // negative number.
465  ptrdiff_t padding = -mem_pd & ( MAX_ALIGN - 1 );
466 
467  // Size calculations differ here from _arena_init
468  // We want a chunk of memory from a l
469 
470 
471  Arena *ap;
472  for ( ap = *p;; *p = ap ) {
473  // Work using a size, not with pointer arithmetic.
474  ptrdiff_t available = ap->end - ap->begin;
475  // assert(available >= 0 );
476 
477  // Per the note, this subtraction is safe. Both operands are
478  // signed, so no surprise promotions turning a small negative
479  // into large positive (i.e. size_t).
480  if ( mem_pd > available - padding ) {
481  if ( ap->next ) {
482  ap = ap->next;
483  ap->begin = ( char * ) ap + _AHS; // reset!
484  // ap->end is never touched!
485  continue; // keep looking
486 
487  } else { // End of the list, allocate a new chunk.
488  // It is *not* yet safe to add header_size to mem_pd,
489  // so subtract from the other side.
490  if ( mem_pd > (PTRDIFF_MAX - _AHS - padding) ) {
491  return NULL; // request too large for metadata
492  }
493  // At this point we know header_size+mem_pd+padding is
494  // safe to compute.
496 
497  // Note: chunk_sz does not require any alignment padding,
498  // accounted for.
499  ptrdiff_t real_size = MAX( ( mem_pd + padding + _AHS ), ( ptrdiff_t ) first[n].chunk_sz );
500 
501  /* ptrdiff_t chunk_sz = ap->chunk_sz; // save a copy */
502 #if 0 == 1
503  fprintf(stderr,"REAL_SIZE: %ld\n",real_size) ;
504 #endif
505 
506  if ( real_size > (ssize_t)ARENAS_MAX_ALLOC ) {
507  fprintf( stderr, alloc_emsg2,"_alloc",real_size, ARENAS_MAX_ALLOC );
508  abort( );
509  } else if ( tot_mem_usage > ARENAS_MAX_ALLOC - real_size ) {
510  fprintf( stderr, alloc_emsg3, "_alloc",real_size, ARENAS_MAX_ALLOC );
511  abort( );
512  }
513 
514  ap = ap->next = malloc( (size_t)real_size );
515  if ( !ap ) {
516  return NULL; // OOM (can happen on Linux with huge mem_sz!)
517  }
518  tot_mem_usage += real_size ; // updates total allocated.
519  if ( real_size < _128K ) {
520  arenas_mem_malloced[n] += real_size ;
521  } else {
522  arenas_mem_mmapped[n] += real_size ;
523  }
524 #if ARENAS_LOG_LEVEL > 0
525 #ifndef CORE_ARENA_NO_LOGGING
526  allocated_chunks[n] += real_size;
527  allocation_chunk_count[n] += 1 ;
528 #endif
529 #endif
530  ap->next = NULL;
531  *p = ap ; // BUGFIX we are breaking out, and won't update in for loop.
532  if ( real_size > (ptrdiff_t) first[n].chunk_sz ) {
533  ap->chunk_sz = (size_t) (real_size - _AHS) ;
534  } else {
535  ap->chunk_sz = first[n].chunk_sz;
536  }
537  ap->begin = ( char * ) ap + _AHS;
538  ap->end = ap->begin + ap->chunk_sz;
539  break; // use this arena
540  }
541  } else {
542  break; // found space
543  }
544  }
545 
546  void *ptr = ap->begin;
547  ap->begin += mem_pd + padding; // checks passed, so addition is safe
548  // Starting point for next memory allocation.
549 
550  // Note: I strongly recommend zero-initialization by default. It
551  // makes for clearer, shorter, and more robust programs. If it's
552  // too slow in some cases, add an opt-out flag.
553 
554  // Zeroing out the last byte and the padding
555  // will implement an `arena_calloc` that zeroes out all returned memory,
556  // to follow the usual convention, to get it done! makes swapping methods
557  // easier.
558  for ( char *zptr = ap->begin - ( padding + 1 ); zptr < ap->begin; zptr++ ) {
559  *zptr = '\0';
560  }
561  return ptr;
562  // return memset(ptr, 0, mem_pd);
563 }

668 {

670

673  abort( ); // Overflow conditions.
674  }
675 
676  static const char *emsg = "arena_alloc: Couldn't allocate memory for arena with mem_pd: %lu.\n" ;
677  // First reject anything nonsensical or excessively large .
678  if ( mem_sz == 0 || mem_sz > PTRDIFF_MAX ) {
679  return NULL; // request impossibly large (out of memory)
680  }
681 
682  void *p;
683  p = arenas[n]->begin; // start of buffer to allocate.
684  ptrdiff_t mem_pd = mem_sz;
685  ptrdiff_t padding = -mem_pd & ( MAX_ALIGN - 1 );
686  mem_pd += padding;
688  if ( mem_pd < ( ptrdiff_t ) mem_sz || ( mem_pd > (PTRDIFF_MAX - _AHS) ) ) {
689  fprintf( stderr, emsg, ( size_t ) mem_pd );
690  abort( ); // Overflow conditions.
691  }
692  ptrdiff_t gauge = (ptrdiff_t)arenas[n]->begin + mem_pd ;
693  assert( gauge >= 0) ;
694  if ( gauge > (ptrdiff_t) arenas[n]->end ) {
695  // padding is already added to mem_pd here.
696  p = _alloc( &arenas[n], mem_pd, n );
697  } else { // zero out last byte and padding
698 #if 0
699  ptrdiff_t padding2 = arenas[n]->begin ;
700  padding2 = -padding2 & ( MAX_ALIGN - 1 );
701  arenas[n]->begin += padding2;
702 #endif
703  arenas[n]->begin += (mem_pd + padding);
704 #if 0
705  for ( char *zptr = arenas[n]->begin - ( padding + 1 ); zptr < arenas[n]->begin; zptr++ ) {
706  *zptr = '\0';
707  }
708 #endif
709  }
710 
711 #if ARENAS_LOG_LEVEL > 1
712  allocated_memory[n] += mem_sz;
713  allocation_memory_count[n] += 1;
714 #endif
715 
716  memset( p, 0, (size_t) mem_pd );
717  /* memset( p, 0, (size_t) (mem_pd + padding) ); */
